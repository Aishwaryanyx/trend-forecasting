model = genai.GenerativeModel("gemini-1.5-flash")


system_prompt = "You are a fashion trend forecaster.You are supposed to predict the future fashion trends by analysing the past and current data of the fashion trends."


# Define the function to interact with the bot
def chat_with_bot(fashion_type, user_input):
    # Adjust the prompt based on the type of fashion the user asks for
    prompt = system_prompt + f"\nFashion type: {fashion_type}\nUser: {user_input}\n:"
    
    # Using the correct method to generate a response from the model
    response = model.generate_content(prompt)
    
    # Limit the response to 10 lines maximum by splitting and trimming
    lines = response.text.split('\n')
    response_text = '\n'.join(lines[:10])  # Keep the first 10 lines
    
    return response_text
